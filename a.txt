#include <iostream>
using namespace std;
//iterativo->falta hacer
int fibonacci(int x)
{
  int arr[x+1];
  arr[0] = 0; arr[1] = 1;
  for(int i=2;i<=x;i++)
  {
    if(!arr[i])fibonacci(arr[i]);
    int n = arr[i-1]+arr[i-2];
    arr[0] = arr[1];
    arr[1] = n;
  }
  return arr[x+1];
}
//recursivo->hecho
int fibonacci2(int x)
{
  if(x==0)return 0;
  if(x==1)return 1;
  return fibonacci2(x-1)+fibonacci2(x-2);
}
